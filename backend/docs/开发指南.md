## 开发环境设置

### 1. 克隆项目

```bash
git clone <repository-url>
cd backend
```

### 2. 安装依赖

```bash
npm install
```

### 3. 配置环境变量

```bash
cp .env.example .env
vim .env
```

### 4. 启动基础设施

```bash
# 使用 Docker 启动 MySQL 和 Redis
docker-compose up -d mysql redis

# 等待服务就绪
sleep 10

# 初始化数据库
mysql -h localhost -P 3306 -u root -p < ../database/lynshae_init.sql
```

### 5. 启动开发服务器

```bash
# 开发模式（自动重启）
npm run dev

# 或者使用 nodemon
nodemon src/server.js
```

### 6. 验证安装

```bash
# 健康检查
curl http://localhost:3000/health

# 预期响应
{"status":"ok","timestamp":"...","uptime":123,"environment":"development"}
```

## 代码规范

### 目录结构约定

```
src/
├── config/          # 配置文件
├── controllers/     # 控制器（业务逻辑）
├── middleware/      # 中间件
├── routes/          # 路由定义
├── services/        # 服务层（复杂业务）
├── models/          # 数据模型
├── utils/           # 工具函数
└── server.js        # 入口文件
```

### 命名规范

```javascript
// 文件名：小写 + 连字符
// user-controller.js (❌)
// userController.js (✅)

// 类名：大驼峰
class UserService {}

// 函数/变量：小驼峰
const getUserById = async () => {}
const userName = 'test'

// 常量：大写 + 下划线
const JWT_SECRET = process.env.JWT_SECRET
const MAX_RETRY_COUNT = 3

// 数据库表：小写 + 下划线 + 复数
// users, products, order_items
```

### 代码风格

```javascript
// 使用 async/await 而不是回调
async function getUser(id) {
  const user = await query('SELECT * FROM users WHERE id = ?', [id]);
  return user[0];
}

// 统一错误处理
try {
  // 业务逻辑
} catch (error) {
  logger.error('操作失败', { error: error.message });
  next(error);
}

// 使用解构
const { id, username } = req.user;

// 使用模板字符串
const message = `Hello, ${username}!`;
```

 * @throws {Error} 用户不存在时抛出

## 开发流程

### 1. 创建新分支

```bash
git checkout -b feature/your-feature-name
```

### 2. 开发功能

遵循 TDD（测试驱动开发）：
1. 编写测试用例
2. 实现功能代码
3. 运行测试
4. 重构优化

### 3. 提交代码

```bash
# 添加更改
git add .

# 提交（遵循约定式提交）
git commit -m "feat: 添加用户登录功能"
# 或
git commit -m "fix: 修复商品列表分页 bug"
```

### 提交类型

| 类型 | 说明 |
|------|------|
| feat | 新功能 |
| fix | 修复 bug |
| docs | 文档更新 |
| style | 代码格式 |
| refactor | 重构 |
| test | 测试相关 |
| chore | 构建/工具 |

### 4. 推送与 PR

```bash
git push origin feature/your-feature-name
# 然后在 GitHub 创建 Pull Request
```

## 测试指南

### 测试框架

使用 Jest 作为测试框架：

```bash
# 运行所有测试
npm test

# 监听模式
npm run test:watch

# 生成覆盖率报告
npm test -- --coverage
```

### 单元测试示例

```javascript
// tests/auth.test.js
const request = require('supertest');
const app = require('../src/server');

describe('认证接口', () => {
  describe('POST /api/v1/auth/login', () => {
    it('应该成功登录', async () => {
      const response = await request(app)
        .post('/api/v1/auth/login')
        .send({
          email: 'test@example.com',
          password: 'password123'
        });

      expect(response.status).toBe(200);
      expect(response.body.code).toBe('SUCCESS');
      expect(response.body.data).toHaveProperty('tokens');
    });

    it('应该拒绝错误的密码', async () => {
      const response = await request(app)
        .post('/api/v1/auth/login')
        .send({
          email: 'test@example.com',
          password: 'wrongpassword'
        });

      expect(response.status).toBe(401);
      expect(response.body.code).toBe('INVALID_CREDENTIALS');
    });
  });
});
```

### 集成测试示例

```javascript
// tests/integration/product.test.js
describe('商品管理', () => {
  let authToken;

  beforeAll(async () => {
    // 登录获取 token
    const loginRes = await request(app)
      .post('/api/v1/auth/login')
      .send({ email: 'admin@lynshae.com', password: 'admin123' });
    authToken = loginRes.body.data.tokens.accessToken;
  });

  it('应该创建商品', async () => {
    const response = await request(app)
      .post('/api/v1/products')
      .set('Authorization', `Bearer ${authToken}`)
      .field('name', '测试商品')
      .field('price', 99.99);

    expect(response.status).toBe(201);
  });
});
```

## 性能优化

### 1. 数据库查询优化

```javascript
// ❌ 避免 N+1 查询
const users = await query('SELECT * FROM users');
for (const user of users) {
  const orders = await query('SELECT * FROM orders WHERE user_id = ?', [user.id]);
}

// ✅ 使用 JOIN
const results = await query(`
  SELECT u.*, o.id as order_id, o.total_amount
  FROM users u
  LEFT JOIN orders o ON u.id = o.user_id
`);
```

### 2. 添加索引

```sql
-- 为常用查询字段添加索引
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_orders_user_id ON orders(user_id);
CREATE INDEX idx_products_category ON products(category_id, is_on_sale);
```

### 3. 使用缓存

```javascript
const { cache, keys } = require('../config/redis');

async function getProduct(id) {
  // 先查缓存
  let product = await cache.get(keys.product(id));
  
  if (!product) {
    // 查数据库
    const products = await query('SELECT * FROM products WHERE id = ?', [id]);
    product = products[0];
    
    // 写入缓存（1 小时）
    await cache.set(keys.product(id), product, 3600);
  }
  
  return product;
}
```

### 4. 分页查询

```javascript
// 永远不要一次性加载所有数据
const page = parseInt(req.query.page) || 1;
const limit = parseInt(req.query.limit) || 20;
const offset = (page - 1) * limit;

const items = await query(
  'SELECT * FROM products LIMIT ? OFFSET ?',
  [limit, offset]
);
```

## 安全最佳实践

### 1. 输入验证

```javascript
const { body, validationResult } = require('express-validator');

router.post('/register', [
  body('email').isEmail().normalizeEmail(),
  body('password').isLength({ min: 6 }),
  body('username').trim().isLength({ min: 3, max: 20 })
], async (req, res) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }
  // 处理逻辑
});
```

### 2. SQL 注入防护

```javascript
// ❌ 禁止字符串拼接
const sql = `SELECT * FROM users WHERE email = '${email}'`;

// ✅ 使用参数化查询
const users = await query('SELECT * FROM users WHERE email = ?', [email]);
```

### 3. XSS 防护

```javascript
// 转义用户输入
const escapeHtml = (str) => {
  const map = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#039;'
  };
  return str.replace(/[&<>"']/g, m => map[m]);
};
```

### 4. 密码安全

```javascript
const bcrypt = require('bcryptjs');
const BCRYPT_ROUNDS = 12;

// 加密密码
const passwordHash = await bcrypt.hash(password, BCRYPT_ROUNDS);

// 验证密码
const isMatch = await bcrypt.compare(password, passwordHash);
```

### 5. 速率限制

```javascript
const rateLimit = require('express-rate-limit');

const limiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 分钟
  max: 100, // 最多 100 个请求
  message: '请求过于频繁'
});

app.use('/api/', limiter);
```
